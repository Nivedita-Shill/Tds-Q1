Data extraction (example)


# sample: compute YoY growth
import pandas as pd

def yoy_growth(df, period_col='period', value_col='revenue'):
    df = df.sort_values(period_col)
    prev = df[value_col].shift(4)  # previous year (quarterly)
    df['yoy_growth'] = (df[value_col] - prev) / prev * 100
    return df
